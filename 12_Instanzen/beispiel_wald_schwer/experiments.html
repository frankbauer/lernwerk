Ändern Sie die Klasse Baum so, dass die Jahrezeit nicht unkontrolliert verändert werden kann. Das Attribut jahreszeit muss private gemacht werden. Außerdem müssen neue Methoden zum Einstellen der Jahreszeit erstellt werden. 
private String jahreszeit = "herbst";

public void setSeason(String jahreszeit){
    //sofern fuer den Uebergabeparameter "jahreszeit"
    //ein gueltiger String uebergeben wird (z.B. "winter"),
    if(jahreszeit.equals("winter")||jahreszeit.equals("fruehling")||
       jahreszeit.equals("sommer")||jahreszeit.equals("herbst")||
       jahreszeit.equals("spaetherbst"))
    {
        //wird in das Instanzattribut "jahreszeit" der Instanz, 
        //auf der die Methode aufgerufen wurde, der uebergebene 
        //String geschrieben
        this.jahreszeit = jahreszeit;

        //Achtung: hier "this" vor dem ersten "jahreszeit" notwendig,
        //damit das Instanzattribut verwendet wird und nicht die
        //lokale Variable (hier der Uebergabeparameter)
        //"jahreszeit = jahreszeit" wuerde dies nicht erfuellen
    }
    //keine Aenderung des Attributs sofern eine
    //ungueltige Jahreszeit uebergeben wird
}

public void getSeason(){
    return this.jahreszeit;
}

//oder fuer jede Jahreszeit separat:
public void setWinter(){
    this.jahreszeit = "winter";
}

public boolean isWinter(){
    return this.jahreszeit.equals("winter");
}

public void setAutumn(){
    this.jahreszeit = "herbst";
}
