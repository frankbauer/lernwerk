Schreiben Sie eine Methode isCyclic die überprüft, ob ein Graph zyklisch ist. Ein Graph ist dann zyklisch, wenn es einen Pfad gibt der bei einem Knoten beginnt und bei demselben Knoten endet. Wenn ein Knoten mehr als einmal besucht wird, ist dies noch kein Zyklus! Sie müssen eine visited-Liste mitführen in der ein Knoten vor dem Besuchen der Kindknoten eingefügt, und nach dem Besuch aller Kindknoten wieder entfernt wird. 
static boolean isCyclic(
            Node<Integer> cur, 
            java.util.List<Node<Integer>> path
) {
    if (cur == null) { return false; }

    //Knoten ist bereits teil des aktuellen pfades in path
    if (path.contains(cur)) return true;
    
    //Knoten als Station auf dem Pfad aufnehmen
    path.add(cur); 

    //Alle Kindknoten prüfen
    for (Node<Integer> child : cur.childNodes()) {        
        if (isCyclic(child, path)) return true;
    }
    //Knoten wieder aus dem Pfad entfernen
    path.remove(cur);

    return false;
}
